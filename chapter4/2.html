vector

vector 是 C++ 标准库中定义的一个类模板

  • 与内建数组相比，更侧重于易用性
    • 可复制、可在运行期动态改变元素个数
  • 构造与初始化
    • 聚合初始化
    • 其它的初始化方式
#include <iostream>
#include <vector>

int main()
{
    std::vector<int> x = {1, 2, 3};
    std::vector<int> y(3);
    y = x;

    std::vector<int> z1 = {1, 1, 1};
    std::vector<int> z2(3, 1); // = z1
}
  • 其它方法
    • 获取元素个数、判断是否为空
    • 插入、删除元素
    • vector 的比较
#include <iostream>
#include <vector>

int main()
{
    std::vector<int> x = {1, 2, 3};
    std::cout << x.size() << std::endl; // 3
    std::cout << x.empty() << std::endl; // 0

    x.push_back(4);
    std::cout << x.size() << std::endl; // 4

    x.pop_back();

    std::vector<int> x1 = {1, 2, 3};
    std::cout << (x == x1) << std::endl; // 1
    std::cout << (x < x1) << std::endl; // 0
}

vector 中元素的索引与遍历

  • [] V.S. at
  • (c)begin / (c)end 函数 V.S. (c)begin / (c)end 方法
#include <iostream>
#include <vector>

int main()
{
    std::vector<int> x = {1, 2, 3};
    std::cout << x[1] << std::endl; // 2
    std::cout << x.at(1) << std::endl; // 2

    auto a = std::begin(x);
    auto b = x.begin();

    while (b != x.end())
    {
        std::cout << *b << std::endl;
        b += 1;
    }

    for (auto val : x)
        std::cout << val << std::endl;
}

vector 相关的其它内容

  • 添加元素可能使迭代器失效
  • 多维 vector
#include <iostream>
#include <vector>

int main()
{
    std::vector<std::vector<int>> x;
    x.push_back(std::vector<int>());
    x[0].push_back(1);
    std::cout << x[0][0] << std::endl; // 1

    std::vector<std::vector<int>> y{{1, 2, 3}, {4, 5, 6}};
    std::cout << y[1][2] << std::endl; // 6
}
  • 从 . 到 -> 操作符
#include <iostream>
#include <vector>

int main()
{
    std::vector<int> x;
    std::cout << x.size() << std::endl;

    std::vector<int>* ptr = &x;
    std::cout << (*ptr).size() << std::endl;
    std::cout << ptr -> size() << std::endl;
}
